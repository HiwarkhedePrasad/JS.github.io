FOR Loop Syntax

for (expression 1; expression 2; expression 3) {
  // code block to be executed
}

FOR IN Loop Syntax

for (key in object) {
  // code block to be executed
}

WHILE Loop Syntax

while (condition) {
  // code block to be executed
}

DO WHILE Loop Syntax

do {
  // code block to be executed
} while (condition);

IF Syntax

if (condition) {
  // code block to be executed if condition is true
}

IF ELSE Syntax

if (condition) {
  // code block to be executed if condition is true
} else {
  // code block to be executed if the condition is false
}

ELSE IF Syntax

if (condition 1) {
  // code block to be executed if condition 1 is true
} else if (condition 2) {
  // code block to be executed if condition 2 is true
} else {
  // code block to be executed if all conditions are false
}

NESTED IF ELSE Syntax

if (firstCondition) {
  // This code executes when the first condition is true
  if (secondCondition) {
    // This code executes when the second condition is also true
  } else {
    // This code executes when only the first condition is true
  }
} else {
  // This code executes when the first condition is false
}

SWITCH CASE Statement Syntax

switch (expression) {
  case x:
    // code block
    break;
  case y:
    // code block
    break;
  default:
    // code block
}

BREAK Statement

The `break` statement immediately terminates the current loop or switch statement and transfers control to the statement following the terminated statement.

break;

CONTINUE Statement

The `continue` statement terminates execution of the statements in the current iteration of the current or labeled loop, and continues with the next iteration of the loop.

continue;
